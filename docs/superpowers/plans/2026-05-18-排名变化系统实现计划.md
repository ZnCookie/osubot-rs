# osubot-rs 完整排名变化系统

## Context

实现历史数据存储 + 排名变化对比系统：
- 后台定时任务按活跃度动态更新用户数据
- 快照存储，查询时取36小时内最近一条对比
- 群内发言触发更新（1小时冷却）

```rust
#[derive(Debug, Clone)]
pub struct UserChange {
    pub rank_change: Option<i64>,
    pub country_rank_change: Option<i64>,
    pub playcount_change: Option<i64>,
    pub hits_change: Option<i64>,
    pub playtime_change: Option<i64>,
}

impl UserChange {
    pub fn has_changes(&self) -> bool {
        self.rank_change.map_or(false, |c| c != 0)
            || self.country_rank_change.map_or(false, |c| c != 0)
            || self.playcount_change.map_or(false, |c| c != 0)
            || self.hits_change.map_or(false, |c| c != 0)
            || self.playtime_change.map_or(false, |c| c != 0)
    }
}

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum UserActivity {
    Active,       // 新增游玩记录（最后4h有新记录）
    SemiActive,   // 有recent但无新增（最后4h有记录但无新增）
    Normal,       // 有快照无变化（最后8h内更新过但无变化）
    Inactive,     // 无变化（最后48h内无更新）
    NoRecent,     // 没变化（最后4-8h有快照但无变化）
    UserNotExists,
}

#[derive(Debug, Clone)]
pub struct UpdateResult {
    pub activity: UserActivity,
    pub added_snapshot: bool,
    pub added_records: i32,
}
```

### get_closest_snapshot_to_hours_ago 实现逻辑

```rust
/// 获取距离 target_hours_ago 最近的快照（用于变化对比）
/// 从 now 前 max_lookback 小时内所有快照中，选择时间差绝对值最小的那一条。
/// 不做任何窗口限制，保证 2 小时内的快照也会被考虑。
pub fn get_closest_snapshot_to_hours_ago(
    &self,
    username: &str,
    mode: GameMode,
    target_hours_ago: i64,
    max_lookback: i64,
) -> SqlResult<Option<(DateTime, UserStats)>> {
    let now = Utc::now();
    let target_time = now - chrono::Duration::hours(target_hours_ago);
    let earliest = now - chrono::Duration::hours(max_lookback);

    // 获取 max_lookback 小时内所有快照
    let all = self.get_snapshots_within_hours(username, mode, max_lookback)?;

    // 过滤出 earliest 之后的快照，按时间差绝对值排序
    let candidates: Vec<_> = all.into_iter()
        .filter(|(_, dt)| *dt >= earliest)
        .collect();

    if candidates.is_empty() {
        return Ok(None);
    }

    Ok(Some(candidates.into_iter().min_by_key(|(_, dt)| {
        (*dt - target_time).num_seconds().unsigned_abs() as i64
    }).unwrap()))
}
```

### 活跃度判定逻辑

```rust
async fn eval_activity(&self, username: &str, mode: GameMode) -> UpdateResult {
    // 1. 获取当前用户数据
    let current = match api::fetch_user_stats(&self.api_key, &self.client_id, username, mode).await {
        Ok(stats) => stats,
        Err(ApiError::NotFound) => {
            return Ok(UpdateResult { activity: UserActivity::UserNotExists, added_snapshot: false, added_records: 0 });
        }
        Err(_) => {
            return Ok(UpdateResult { activity: UserActivity::Inactive, added_snapshot: false, added_records: 0 });
        }
    };

    // 2. 获取最近快照
    let latest = self.storage.get_latest_snapshot(username, mode)?;

    // 3. 【提前判定 Inactive】PlayCount 相同 且 PP == 0
    if let Some(ref snap) = latest {
        if current.playcount == snap.playcount && current.pp == 0.0 {
            return Ok(UpdateResult { activity: UserActivity::Inactive, added_snapshot: false, added_records: 0 });
        }
    }

    // 4. 数据有变化则保存新快照（先保存再计算变化，保证基准最新）
    let stats_changed = latest.as_ref().map_or(true, |l| l.rank != current.rank || l.pp != current.pp);
    let mut added_snapshot = false;
    if stats_changed {
        self.storage.save_stats(username, mode, &current)?;
        added_snapshot = true;
    }

    // 5. 获取 recent 游玩记录，先尝试写入数据库获取实际新增数量
    let recent_plays = api::get_user_recent(&self.api_key, &self.client_id, username, mode).await?;
    let added_records = self.storage.save_play_records(username, mode, &recent_plays)?;

    if added_records > 0 {
        // 有实际新增记录 → Active
        return Ok(UpdateResult { activity: UserActivity::Active, added_snapshot, added_records });
    }

    // 6. 检查最后4h是否有记录（无新增）
    let now = Utc::now();
    let has_recent = recent_plays.iter().any(|p| (now - p.beatmap.lastplayed).num_hours() < 4);
    if has_recent {
        return Ok(UpdateResult { activity: UserActivity::SemiActive, added_snapshot, added_records: 0 });
    }

    // 7. 计算变化（基于刚保存的新快照或旧快照）
    let change = self.storage.calculate_change(username, mode, &current)?;

    // 8. 检查最后更新时间
    let last_update = self.storage.get_last_update(username, mode)?;
    let hours_since_update = last_update.map(|t| (now - t).num_hours()).unwrap_or(i64::MAX);

    if change.map(|c| c.has_changes()).unwrap_or(false) {
        // 有变化
        if hours_since_update < 4 {
            return Ok(UpdateResult { activity: UserActivity::NoRecent, added_snapshot: true, added_records: 0 });
        }
        return Ok(UpdateResult { activity: UserActivity::Normal, added_snapshot: true, added_records: 0 });
    }

    // 无变化
    if hours_since_update < 8 {
        return Ok(UpdateResult { activity: UserActivity::NoRecent, added_snapshot: false, added_records: 0 });
    }
    if hours_since_update < 48 {
        return Ok(UpdateResult { activity: UserActivity::Normal, added_snapshot: false, added_records: 0 });
    }
    Ok(UpdateResult { activity: UserActivity::Inactive, added_snapshot: false, added_records: 0 })
}

fn get_update_interval(&self, activity: UserActivity) -> Duration {
    match activity {
        UserActivity::Active => Duration::hours(self.config.active_interval_hours),       // 2h
        UserActivity::SemiActive => Duration::hours(self.config.semi_active_interval_hours), // 4h
        UserActivity::Normal => Duration::hours(self.config.normal_interval_hours),       // 8h
        UserActivity::Inactive => Duration::hours(self.config.inactive_interval_hours),   // 48h
        UserActivity::NoRecent => Duration::hours(6),                                      // 6h
        UserActivity::UserNotExists => Duration::hours(24),
    }
}

fn update_next_time(&self, username: &str, mode: GameMode, activity: UserActivity) {
    let interval = self.get_update_interval(activity);
    let next = Utc::now() + interval;
    self.storage.set_next_update(username, mode, next).ok();
}

async fn process_due_users(&self) {
    let now = Utc::now();
    let due = match self.storage.get_due_users() {
        Ok(d) => d,
        Err(_) => return,
    };

    for (username, mode) in due {
        let result = self.eval_activity(&username, mode).await;
        self.update_next_time(&username, mode, result.activity);
    }
}

pub async fn run(&self) {
    loop {
        tokio::time::sleep(tokio::time::Duration::from_secs(60 * self.config.interval_minutes)).await;
        self.process_due_users().await;
    }
}
```
