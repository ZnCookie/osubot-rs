```rust
pub enum GameMode {

impl GameMode {
    pub fn from_str(s: &str) -> Option<GameMode> {
        match s.trim() {
            "0" | "" => Some(GameMode::Osu),
            "1" => Some(GameMode::Taiko),
            "2" => Some(GameMode::Catch),
            "3" => Some(GameMode::Mania),
            _ => None,
        }
    }

    pub fn name(&self) -> &'static str {
        match self {
            GameMode::Osu => "osu!",
            GameMode::Taiko => "osu!taiko",
            GameMode::Catch => "osu!catch",
            GameMode::Mania => "osu!mania",
        }
    }

    pub fn api_value(&self) -> &'static str {
        match self {
            GameMode::Osu => "osu",
            GameMode::Taiko => "taiko",
            GameMode::Catch => "fruits",
            GameMode::Mania => "mania",

    }
}
```

```rust
use crate::types::{Command, GameMode};

/// 解析用户消息为命令
/// 支持格式:
/// - `~` / `~0` - 查询自己 std
/// - `~1` / `~,1` - 查询自己 taiko
/// - `~2` / `~,2` - 查询自己 catch
/// - `~3` / `~,3` - 查询自己 mania
/// - `where <用户名>` - 查询他人 std
/// - `where <用户名>,<模式>` - 查询他人指定模式
/// - `查@<QQ用户>` - 查询他人 std
/// - `查@<QQ用户>,<模式>` - 查询他人指定模式
/// - `绑定 <osu用户名>` - 绑定账号
/// - `解绑` - 解绑账号
pub fn parse_command(msg: &str) -> Option<Command> {
    let msg = msg.trim();

    // 查询自己: ~ 或 ~<模式>
    if msg.starts_with('~') {
        let rest = msg.trim_start_matches('~').trim_start_matches(',');
        if rest.is_empty() {
            return Some(Command::QuerySelf { mode: GameMode::Osu });
        }
        let mode = GameMode::from_str(rest)?;
        return Some(Command::QuerySelf { mode });
    }

    // 查询他人: where <用户名> [, 模式]
    if let Some(rest) = msg.strip_prefix("where ") {
        let parts: Vec<&str> = rest.split(',').collect();
        let username = parts[0].trim().to_string();
        let mode = if parts.len() > 1 {
            GameMode::from_str(parts[1].trim())?
        } else {
            GameMode::Osu
        };
        return Some(Command::QueryUser { username, mode });
    }

    // 查询他人: 查@<QQ用户> [, 模式]
    if let Some(rest) = msg.strip_prefix("查@") {
        let parts: Vec<&str> = rest.split(',').collect();
        let username = parts[0].trim().to_string();
        let mode = if parts.len() > 1 {
            GameMode::from_str(parts[1].trim())?
        } else {
            GameMode::Osu
        };
        return Some(Command::QueryUser { username, mode });
    }

    // 绑定: 绑定 <osu用户名>
    if let Some(username) = msg.strip_prefix("绑定 ") {
        let username = username.trim();
        if username.is_empty() {
            return None;
        }
        return Some(Command::Bind { username: username.to_string() });
    }

    // 解绑
    if msg == "解绑" {
        return Some(Command::Unbind);
    }

    None
}
```

```rust
use crate::types::{GameMode, UserStats};
use reqwest::Client;
use thiserror::Error;

#[derive(Error, Debug)]
pub enum ApiError {
    #[error("HTTP request failed: {0}")]
    Http(#[from] reqwest::Error),
    #[error("User not found")]
    NotFound,
    #[error("Invalid API response")]
    InvalidResponse,
    #[error("API key missing")]
    MissingApiKey,
}

#[derive(Debug, serde::Deserialize)]
struct OsuUserResponse {
    username: String,
    #[serde(rename = "pp_raw")]
    pp: Option<String>,
    #[serde(rename = "pp_rank")]
    rank: Option<String>,
    #[serde(rename = "pp_country_rank")]
    country_rank: Option<String>,
    #[serde(rename = "ranked_score")]
    ranked_score: Option<String>,
    #[serde(rename = "accuracy")]
    accuracy: Option<String>,
    #[serde(rename = "playcount")]
    playcount: Option<String>,
    #[serde(rename = "total_hits")]
    hits: Option<String>,
    #[serde(rename = "total_playtime")]
    playtime: Option<String>, // minutes
    #[serde(rename = "rank_change")]
    rank_change: Option<serde_json::Value>,
}

#[derive(Debug, serde::Deserialize)]
struct OsuApiResponse {
    statistics: Option<OsuUserResponse>,
}

/// 从 osu! API v2 获取用户数据
pub async fn fetch_user_stats(
    api_key: &str,
    username: &str,
    mode: GameMode,
) -> Result<UserStats, ApiError> {
    if api_key.is_empty() {
        return Err(ApiError::MissingApiKey);
    }

    let client = Client::new();
    let mode_param = mode.api_value();

    let url = format!(
        "https://osu.ppy.sh/api/v2/users/{}/statistics/osu",
        username
    );

    let resp = client
        .get(&url)
        .header("Authorization", format!("Bearer {}", api_key))
        .query(&[("mode", mode_param)])
        .send()
        .await?;

    if resp.status() == 404 {
        return Err(ApiError::NotFound);
    }

    let data: OsuApiResponse = resp.json().await?;

    let stats = data.statistics.ok_or(ApiError::InvalidResponse)?;

    let parse_i64 = |s: Option<String>| -> i64 {
        s.and_then(|v| v.replace(',', "").parse().ok())
            .unwrap_or(0)
    };

    let parse_f64 = |s: Option<String>| -> f64 {
        s.and_then(|v| v.replace('%', "").parse().ok())
            .unwrap_or(0.0)
    };

    let rank_change = stats.rank_change.as_ref().and_then(|v| {
        if let Some(obj) = v.as_object() {
            obj.get("all")?.as_i64()
        } else {
            None
        }
    });

    let country_rank_change = stats.rank_change.as_ref().and_then(|v| {
        if let Some(obj) = v.as_object() {
            obj.get("country")?.as_i64()
        } else {
            None
        }
    });

    Ok(UserStats {
        username: stats.username,
        pp: parse_f64(stats.pp),
        rank: parse_i64(stats.rank),
        country_rank: parse_i64(stats.country_rank),
        ranked_score: parse_i64(stats.ranked_score),
        accuracy: parse_f64(stats.accuracy),
        playcount: parse_i64(stats.playcount),
        hits: parse_i64(stats.hits),
        playtime: (parse_i64(stats.playtime) * 60), // 转成秒
        rank_change,
        country_rank_change,
    })
}
```

```rust
use rusqlite::{params, Connection, Result as SqlResult};
use std::path::Path;
use std::sync::Mutex;

pub struct Storage {
    conn: Mutex<Connection>,
}

impl Storage {
    pub fn new<P: AsRef<Path>>(path: P) -> SqlResult<Self> {
        let conn = Connection::open(path)?;
        conn.execute(
            "CREATE TABLE IF NOT EXISTS user_bindings (
                qq INTEGER PRIMARY KEY,
                osu_username TEXT NOT NULL,
                created_at TEXT DEFAULT CURRENT_TIMESTAMP
            )",
            [],
        )?;
        Ok(Self { conn: Mutex::new(conn) })
    }

    pub fn bind(&self, qq: i64, username: &str) -> SqlResult<()> {
        let conn = self.conn.lock().unwrap();
        conn.execute(
            "INSERT OR REPLACE INTO user_bindings (qq, osu_username) VALUES (?1, ?2)",
            params![qq, username],
        )?;
        Ok(())
    }

    pub fn unbind(&self, qq: i64) -> SqlResult<()> {
        let conn = self.conn.lock().unwrap();
        conn.execute("DELETE FROM user_bindings WHERE qq = ?1", params![qq])?;
        Ok(())
    }

    pub fn get_binding(&self, qq: i64) -> SqlResult<Option<String>> {
        let conn = self.conn.lock().unwrap();
        let mut stmt = conn.prepare("SELECT osu_username FROM user_bindings WHERE qq = ?1")?;
        let mut rows = stmt.query(params![qq])?;
        if let Some(row) = rows.next()? {
            Ok(Some(row.get(0)?))
        } else {
            Ok(None)
        }
    }
}
```

```rust
/// 格式化用户数据为响应字符串
pub fn format_stats(stats: &UserStats, mode: GameMode) -> String {
    let username = &stats.username;
    let mode_name = mode.name();

    let pp = format_number(stats.pp, 2);
    let rank = format_number(stats.rank, 0);
    let country_rank = format_number(stats.country_rank, 0);
    let ranked_score = format_number(stats.ranked_score / 1_000_000, 0); // 转成 m
    let acc = format_number(stats.accuracy, 2);
    let playcount = format_number(stats.playcount, 0);
    let hits = format_number(stats.hits, 0);
    let playtime = format_playtime(stats.playtime);

    let rank_change = format_change(stats.rank_change);
    let country_rank_change = format_change(stats.country_rank_change);

    format!(
        "{username}的个人信息—{mode_name}\n\
         {pp}pp 表现\n\
         #{rank} ({rank_change})\n\
         China #{country_rank} ({country_rank_change})\n\
         {ranked_score},000m Ranked谱面总分\n\
         {acc}% 准确率\n\
         {playcount} 游玩次数\n\
         {hits} 总命中次数\n\
         {playtime}游玩时间",
        username = username,
        mode_name = mode_name,
        pp = pp,
        rank = rank,
        rank_change = rank_change,
        country_rank = country_rank,
        country_rank_change = country_rank_change,
        ranked_score = ranked_score,
        acc = acc,
        playcount = playcount,
        hits = hits,
        playtime = playtime,
    )
}

/// 格式化数字，加千位分隔符
fn format_number<T: itoa::Integer>(value: T, decimals: usize) -> String {
    let mut buf = itoa::Buffer::new();
    let formatted = buf.format(value);

    let chars: Vec<char> = formatted.chars().collect();
    let len = chars.len();

    let mut result = String::new();
    for (i, c) in chars.iter().enumerate() {
        if i > 0 && (len - i) % 3 == 0 {
            result.push(',');
        }
        result.push(*c);
    }

    if decimals > 0 && value as f64 != value as i64 as f64 {
        // 对于浮点数，保留小数位
        let float_val = value as f64;
        let int_part = float_val as i64;
        let dec_part = ((float_val - int_part as f64) * 10_f64.powi(decimals as i32)).round() as i64;

        let mut int_buf = itoa::Buffer::new();
        let int_formatted = int_buf.format(int_part);

        let chars: Vec<char> = int_formatted.chars().collect();
        let len = chars.len();

        let mut int_result = String::new();
        for (i, c) in chars.iter().enumerate() {
            if i > 0 && (len - i) % 3 == 0 {
                int_result.push(',');
            }
            int_result.push(*c);
        }

        let mut dec_buf = itoa::Buffer::new();
        let dec_formatted = dec_buf.format(dec_part);

        format!("{}.{:0>width$}", int_result, dec_formatted, width = decimals)
    } else {
        result
    }
}

/// 格式化排名变化
fn format_change(change: Option<i64>) -> String {
    match change {
        Some(c) if c > 0 => format!("↑{}", c),
        Some(c) if c < 0 => format!("↓{}", c.abs()),
        _ => "—".to_string(),
    }
}

/// 格式化游玩时间
fn format_playtime(seconds: i64) -> String {
    let hours = seconds / 3600;
    let minutes = (seconds % 3600) / 60;
    let secs = seconds % 60;
    format!("{} 小时 {} 分钟 {} 秒", hours, minutes, secs)
}
```

用 tokio::net::TcpStream 和 tokio_tungstenite 连接到 OneBot 11:

```rust
mod config;

use config::Config;
use osubot_core::{
    parse_command,
    storage::Storage,
    response::format_stats,
    types::{Command, GameMode},
    api::{self, ApiError},
};
use tokio::net::TcpStream;
use tokio::sync::mpsc;
use tokio_tungstenite::{connect_async, tungstenite::Message};

#[derive(Debug)]
struct QQMessage {
    group_id: i64,
    user_id: i64,
    message: String,
}

async fn connect_onebot(url: &str) -> Result<(), Box<dyn std::error::Error>> {
    let (ws_stream, _) = connect_async(url).await?;
    println!("Connected to OneBot 11");

    let (write, mut read) = ws_stream.split();

    // 命令通道
    let (cmd_tx, mut cmd_rx) = mpsc::channel::<(QQMessage, String, mpsc::Sender<String>)>(100);

    // spawn 消息接收循环
    let read_task = tokio::spawn(async move {
        while let Some(msg) = read.next().await {
            if let Ok(Message::Text(text)) = msg {
                // 解析 OneBot 消息 JSON，提取 group_id, user_id, message
                if let Some(derived) = parse_onebot_message(&text) {
                    let (tx, mut rx) = mpsc::channel(1);
                    let _ = cmd_tx.send((derived, tx.clone())).await;
                }
            }
        }
    });

    // spawn 命令处理循环
    let process_task = tokio::spawn(async move {
        while let Some((msg, resp_tx)) = cmd_rx.recv().await {
            let user_id = msg.user_id;
            let group_id = msg.group_id;
            let text = msg.message.clone();

            tokio::spawn(handle_command(
                user_id,
                group_id,
                text,
                resp_tx,
            ));
        }
    });

    // 保持连接
    read_task.await?;
    process_task.await?;

    Ok(())
}

async fn handle_command(
    user_id: i64,
    _group_id: i64,
    text: String,
    resp_tx: mpsc::Sender<String>,
) {
    let cmd = match parse_command(&text) {
        Some(cmd) => cmd,
        None => return,
    };

    match cmd {
        Command::QuerySelf { mode } => {
            // TODO: 获取用户绑定，然后查询 API
            let _ = resp_tx.send("正在查询...".to_string()).await;
        }
        Command::QueryUser { username, mode } => {
            // TODO: 查询 API
            let _ = resp_tx.send(format!("查询 {} 的 {} 模式数据...", username, mode.name())).await;
        }
        Command::Bind { username } => {
            let _ = resp_tx.send(format!("绑定 {} 成功（暂未实现）", username)).await;
        }
        Command::Unbind => {
            let _ = resp_tx.send("解绑成功（暂未实现）".to_string()).await;
        }
        Command::Help => {
            let help = "osu! 查分机器人命令：\n\
                        ~ 查询自己的 osu! 数据\n\
                        ~1/2/3 查询对应模式\n\
                        where <用户名> 查询他人\n\
                        绑定 <osu用户名>\n\
                        解绑";
            let _ = resp_tx.send(help.to_string()).await;
        }
    }
}

fn parse_onebot_message(json: &str) -> Option<QQMessage> {
    // 简化解析，实际使用 serde_json
    // 需要从 post_type=message, message_type=group 中提取

    None
}

fn main() {
    let config = Config::from_path("osubot.toml")
        .expect("Failed to load config");

    println!("osubot starting...");

    // 启动 WebSocket 连接
    let rt = tokio::runtime::Runtime::new().unwrap();
    rt.block_on(async {
        if let Err(e) = connect_onebot(&config.bot.onebot_url).await {
            eprintln!("Connection error: {}", e);
        }
    });
}
```
